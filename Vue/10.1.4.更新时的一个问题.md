				{id:002,name:'周冬雨',age:19,sex:'女'},
				{id:003,name:'周杰伦',age:21,sex:'男'},

        	updateMei(){
        		this.persons[0].name = '马老师',	//奏效
        		this.persons[0].age = 50			//奏效

        	}
        }
        
    })
