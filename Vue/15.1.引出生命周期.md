### 外置定时器解决

```
<!DOCTYPE html> 
<html>
    <head>
        <meta charset="UTF-8" />
        <title>引出生命周期</title>
        <!--引入Vue -->
        <script type="text/javascript" src="../js/vue.js"</script>
    </head>
    <body>
        <!--准备好一个容器-->
        <div id="root">
        	<h2 style="{opacity}">欢迎学习Vue</h2>
        </div>
    </body>
    <script type="text/javascript">
        Vue.config. productionTip = false //阻止 vue在启动时生成生产提示。
        const vm = new Vue({
            el:'#root', 
            data:{
                opacity: 1,
            },
        })
        setInterval(() => {
        	vm.opacity -= 0.01
        	if(vm.opacity <= 0) vm.opacity = 1
        },16)
        </script>
</html>
```
